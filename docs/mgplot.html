183        del rkwargs[AX]
184

186    counter = 0
187    for start, tag in zip(stags[STARTS], stags[TAGS]):
188        if start and not isinstance(start, pd.Period):
189            start = pd.Period(start, freq=cast(pd.PeriodIndex, start.index).freq)

def calc_growth( series: pandas.core.series.Series) -> tuple[pandas.core.series.Series, pandas.core.series.Series]: 
20def calc_growth(series: Series) -> tuple[Series, Series]:
21    """
22    Calculate annual and periodic growth for a pandas Series,
23    where the index is a PeriodIndex.
24
25    Args:
26    -   series: A pandas Series with an appropriate PeriodIndex.
27
28    Returns a two element tuple:
29    -   annual: A pandas Series with the annual growth rate.
30    -   periodic: A pandas Series with the periodic growth rate.
31
32    Raises
33    -   TypeError if the series is not a pandas Series.
34    -   TypeError if the series index is not a PeriodIndex.
35    -   ValueError if the series is empty.
36    -   ValueError if the series index does not have a frequency of Q, M, or D.
37    -   ValueError if the series index has duplicates.
38    """
39
40    # --- sanity checks
41    if not isinstance(series, Series):
42        raise TypeError("The series argument must be a pandas Series")
43    if not isinstance(series.index, PeriodIndex):
44        raise TypeError("The series index must be a pandas PeriodIndex")
45    if series.empty:
46        raise ValueError("The series argument must not be empty")
47    if series.index.freqstr[0] not in ("Q", "M", "D"):
