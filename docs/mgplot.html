 94    """
 95    Build a single plot from the data passed in.
 96    This can be a single- or multiple-line plot.
 97    Return the axes object for the build.
 98
 99    Agruments:
100    - data: DataFrame | Series - data to plot
101    - kwargs:
102        - ax: plt.Axes | None - axes to plot on (optional)
103        - dropna: bool | list[bool] - whether to delete NAs frm the
104          data before plotting [optional]
105        - color: str | list[str] - line colors.
106        - width: float | list[float] - line widths [optional].
107        - style: str | list[str] - line styles [optional].
108        - alpha: float | list[float] - line transparencies [optional].
109        - marker: str | list[str] - line markers [optional].
110        - marker_size: float | list[float] - line marker sizes [optional].
111        - annotate: bool | list[bool] - whether to annotate a series.
112        - rounding: int | bool | list[int | bool] - number of decimal places
113          to round an annotation. If True, a default between 0 and 2 is

117        - drawstyle: str | list[str] - matplotlib line draw styles.
118
119    Returns:
120    - axes: plt.Axes - the axes object for the plot
121    """
122
123    # sanity checks
124    report_kwargs(called_from="line_plot", **kwargs)
125    data = check_clean_timeseries(data)
126    validate_kwargs(LP_KW_TYPES, called_from="line_plot", **kwargs)
127
128    # the data to be plotted:
129    df = DataFrame(data)  # really we are only plotting DataFrames
130    df, kwargs = constrain_data(df, **kwargs)
131    if df.empty:
132        print("Warning: No data to plot.")
133
134    # get the arguments for each line we will plot ...
135    item_count = len(df.columns)
136    num_data_points = len(df)
137    swce, kwargs = _get_style_width_color_etc(item_count, num_data_points, **kwargs)
138
139    # Let's plot
140    axes, kwargs = get_axes(**kwargs)  # get the axes to plot on
141
142    for i, column in enumerate(df.columns):
143        series = df[column]
144        series = series.dropna() if DROPNA in swce and swce[DROPNA][i] else series
145        if series.empty or series.isna().all():
146            continue
147
148        axes = series.plot(
149            ls=swce[STYLE][i],
150            lw=swce[WIDTH][i],
151            color=swce[COLOR][i],
152            alpha=swce[ALPHA][i],
153            marker=swce[MARKER][i],
154            ms=swce[MARKERSIZE][i],
155            drawstyle=swce[DRAWSTYLE][i],
156            ax=axes,
157        )
158
159        if swce[ANNOTATE][i] is None or not swce[ANNOTATE][i]:
160            continue
161
162        annotate_series(
163            series,
164            axes,
165            rounding=swce[ROUNDING][i],
166            color=swce[COLOR][i],
167            fontsize=swce[FONTSIZE][i],
168        )
169
170    # add a legend if requested
171    if len(df.columns) > 1:
172        kwargs[LEGEND] = kwargs.get(LEGEND, get_setting("legend"))
173    if LEGEND in kwargs and kwargs[LEGEND] is not None:
174        legend = kwargs[LEGEND]
175        if isinstance(legend, bool):

193          defaults to "zscores".
